def load_card_graphics(csv_file):
    cards = []
    with open(csv_file, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            card = {
                'name': row['Nazwa'],
                'graphics_path': row['Grafika']
            }
            cards.append(card)
    return cards
